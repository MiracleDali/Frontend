对象的扩展

    var name = '张三'
    const user = {
        // 属性简写 属性名和属性值是同样的变量名称，就可以省略
        name,
        age: 20,
        sex: '女',

        // 原来的写法
        say:function() {
            console.log('hello world');
        },
        // 方法简写 新的写法
        say() {
            console.log('hello world');
        }
    }
    console.log(user.name)  // 张三
    user.say()  // hello world


    // 属性简写在函数返回值中的用法
    function getpoint() {
        var x = 10;
        var y = 20;
        return { x,y }
    }
    console.log(getpoint())


    // 属性名表达式
    let propkey = 'zhangruonan';
    let people = {
        [propkey]: '张三',
        [propkey + '1']: '张三1'
    }
    console.log(people)


    // 对象的扩展运算符
    let z = { x: 10, y: 20, z: 30 }
    let o = { ...z, z: 33 }
    console.log(o)
